场景参数
scene默认参数
scene: {
    webglrenderer: {
        alpha: true, //是否支持背景透明
    },
    stats: {
        show: true,
        mode: 0, //监控模式 0:画面每秒传输帧数（fps）  1：画面渲染的时间
    },
    background: {
        type: "sky",
        color: {
            alpha: 0.3, //透明度 取值范围0~1
            color: "#944",
        },
        img: {
            url: "",
        },
        sky: {
            color: { top: "#86b6f5", line: "#ffffff", bottom: "#999999" },
            params: {
                radius: 4000,
                widthSegments: 32,
                heightSegments: 15,
                skyCenter: [0, 0, 0],
            },
        },
    },
    fog: {
        show: false,
        cameraView: false, //相机附近视野清晰
        color: "#fff",
    },
}
camera默认参数
camera: {
    position: {
        x: 0,
        y: 80,
        z: 200,
    },
    near: 0.1,
    far: 2000,
}
